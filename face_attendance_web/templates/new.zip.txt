  <script>
    const video = document.getElementById('video');
    const canvas = document.getElementById('canvas');
    const statusDiv = document.getElementById('status');
    const classDiv = document.getElementById('class-display');

    // 1. Start Camera
    navigator.mediaDevices.getUserMedia({ 
        video: { 
            facingMode: "user",
            width: { ideal: 640 },
            height: { ideal: 480 }
        } 
    })
    .then(stream => { 
        video.srcObject = stream; 
        statusDiv.innerText = "🔍 Scanning...";
    })
    .catch(err => { 
        console.error("Camera error:", err);
        statusDiv.innerText = "❌ Camera Blocked! Check Permissions.";
        statusDiv.classList.add("active-red");
    });

    // 2. Loop: Capture Frame & Send to Python
    setInterval(() => {
        if (!video.videoWidth) return; 

        const context = canvas.getContext('2d');
        canvas.width = video.videoWidth;
        canvas.height = video.videoHeight;
        context.drawImage(video, 0, 0, canvas.width, canvas.height);
        
        const dataURL = canvas.toDataURL('image/jpeg', 0.6); // 0.6 quality for speed

        fetch('/process_frame', {
            method: 'POST',
            body: JSON.stringify({ image: dataURL }),

  <script>
    let attendanceRecords = {};

    async function loadStudents() {
      const classId = document.getElementById('subjectSelect').value;
      const date = document.getElementById('attendanceDate').value;
      if (!classId) return;

      const tbody = document.getElementById('studentsBody');
      tbody.innerHTML = '<tr><td colspan="5" style="text-align:center;">Loading students...</td></tr>';

      try {
        const response = await fetch(`/get_class_students/${classId}?date=${date}`);
        const data = await response.json();
        
        tbody.innerHTML = '';
        attendanceRecords = {}; // Reset

        if (data.students.length === 0) {
          tbody.innerHTML = '<tr><td colspan="5" style="text-align:center;">No approved students found in this semester.</td></tr>';
          return;
        }

        data.students.forEach(student => {
          const sid = student.id;
          const status = data.existing_attendance[sid] || ''; 
          attendanceRecords[sid] = status; // Init

          const row = document.createElement('tr');
          row.innerHTML = `

    // 1. Initialize Week Dropdown
    function initWeeks() {
      const select = document.getElementById('weekSelect');
      select.innerHTML = '';
      
      let curr = new Date();
      const day = curr.getDay();
      const diff = curr.getDate() - day + (day == 0 ? -6 : 1);
      curr.setDate(diff);

      for (let i = 0; i < 6; i++) {
        let start = new Date(curr);
        start.setDate(start.getDate() - (i * 7));
        
        let end = new Date(start);
        end.setDate(end.getDate() + 6);
        
        let startStr = start.toLocaleDateString('en-US', { month: 'short', day: 'numeric' });
        let endStr = end.toLocaleDateString('en-US', { month: 'short', day: 'numeric' });
        let label = i === 0 ? "Current Week" : `Week ${i} Ago`;
        
        let option = document.createElement('option');
        option.value = start.toISOString().split('T')[0];
        option.text = `${label} (${startStr} - ${endStr})`;
        select.appendChild(option);
      }
    }

    // 2. Fetch Data
    async function fetchData() {
      const sem = document.getElementById('semesterSelect').value;
      const sub = document.getElementById('subjectSelect').value;
      const startVal = document.getElementById('weekSelect').value;
      
      // Update Table Headers (Mon Oct 23)
      updateHeaders(startVal);

      const tbody = document.getElementById('attendanceGrid');
      tbody.innerHTML = '<tr><td colspan="8" style="padding:40px; color:#888;">🔄 Loading records...</td></tr>';

      try {
        // CALL NEW PROFESSOR API
        const url = `/get_professor_weekly_attendance?semester=${encodeURIComponent(sem)}&subject=${encodeURIComponent(sub)}&start_date=${startVal}`;
        const res = await fetch(url);
        const data = await res.json();

        tbody.innerHTML = '';
        
        if(data.length === 0) {
          tbody.innerHTML = '<tr><td colspan="8" style="padding:40px; color:#666;">No students found in this semester.</td></tr>';
          return;
        }

        let baseDate = new Date(startVal);

        data.forEach(student => {
          let daysHtml = '';
          const days = ['M','T','W','T','F','S'];
          
          for(let i=0; i<6; i++) {
            let loopDate = new Date(baseDate);
            loopDate.setDate(loopDate.getDate() + i);
            let dateStr = loopDate.toISOString().split('T')[0];
            
            let record = student.week_data[dateStr];
            let status = record ? record.status : 'None';
            let tooltip = record ? `${record.status} - ${record.time}` : 'No Class';

            daysHtml += `<td>
              <div class="day-circle status-${status}" data-tooltip="${tooltip}">
                ${days[i]}
              </div>
            </td>`;
          }

          let pct = student.overall_percent;
          let pctClass = pct >= 75 ? 'pct-high' : (pct >= 50 ? 'pct-mid' : 'pct-low');

          const tr = document.createElement('tr');
          tr.innerHTML = `
            <td style="text-align:left; padding-left:20px;">
              <div style="font-weight:bold; color:#fff; font-size:1.1em;">${student.name}</div>
              <div style="font-size:0.85em; color:#888;">${student.roll}</div>
            </td>
            ${daysHtml}
            <td><span class="pct-badge ${pctClass}">${pct}%</span></td>
          `;
          tbody.appendChild(tr);
        });

      } catch(e) {
        console.error(e);
        tbody.innerHTML = '<tr><td colspan="8" style="color:#ff4444;">❌ Error loading data</td></tr>';
      }
    }

        current.setDate(current.getDate() + 1);
      }

  <script>
    // Store attendance state: { studentId: 'Present' | 'Absent' | 'Leave' }
    let attendanceState = {};

    async function loadStudents() {
      const classId = document.getElementById('classSelect').value;
      const date = document.getElementById('dateInput').value;
      const container = document.getElementById('studentContainer');
      
      if(!classId) return;

      container.innerHTML = '<div class="loading-msg">🔄 Loading student list...</div>';
      attendanceState = {};

      try {
        const res = await fetch(`/get_class_students/${classId}?date=${date}`);
        const data = await res.json();
        
        container.innerHTML = '';

        if(data.students.length === 0) {
          container.innerHTML = '<div class="loading-msg">⚠️ No students found in this semester.</div>';
          return;
        }

        data.students.forEach(student => {
          // Check if record exists, otherwise Default to 'Present'
          const savedStatus = data.existing_attendance[student.id] || 'Present';
          attendanceState[student.id] = savedStatus;

          const card = document.createElement('div');
          card.className = 'student-card';
          card.innerHTML = `
            <div class="student-info">
              <div class="s-name">${student.name}</div>
              <div class="s-roll">${student.roll_no}</div>
            </div>
            <div class="status-options">
              <div class="circle-btn btn-p ${savedStatus === 'Present' ? 'active' : ''}" 
                   onclick="setStatus(${student.id}, 'Present', this)">P</div>
                   
              <div class="circle-btn btn-a ${savedStatus === 'Absent' ? 'active' : ''}" 
                   onclick="setStatus(${student.id}, 'Absent', this)">A</div>
                   
              <div class="circle-btn btn-l ${savedStatus === 'Leave' ? 'active' : ''}" 
                   onclick="setStatus(${student.id}, 'Leave', this)">L</div>
            </div>
          `;
          container.appendChild(card);
        });

      } catch(err) {
        console.error(err);
        container.innerHTML = '<div class="loading-msg" style="color:#ff4444">❌ Error loading data.</div>';
      }
    }

    // Handle Circle Click
    function setStatus(studentId, status, element) {
      // 1. Update Logic State
      attendanceState[studentId] = status;

      // 2. Update Visuals
      const parent = element.parentElement;
      
      // Remove 'active' from all siblings
      const siblings = parent.querySelectorAll('.circle-btn');
      siblings.forEach(btn => btn.classList.remove('active'));

      // Add 'active' to clicked circle
      element.classList.add('active');
    }

    async function saveAttendance() {
      const classId = document.getElementById('classSelect').value;
      const date = document.getElementById('dateInput').value;
      
      if(!classId) {
        alert("Please select a class first!");
        return;
      }

      // Show temporary loading text on button
      const btn = document.querySelector('.save-btn');
      const originalText = btn.textContent;
      btn.textContent = "⏳ Saving...";

      try {
        const res = await fetch('/save_manual_attendance', {
          method: 'POST',
          headers: {'Content-Type': 'application/json'},
          body: JSON.stringify({ 
            class_id: classId, 
            date: date, 
            attendance: attendanceState 
          })
        });
        
        const result = await res.json();
        
        if(result.success) {
          alert("✅ Attendance Saved Successfully!");
        } else {
          alert("❌ Error: " + result.error);
        }
      } catch(e) {
        console.error(e);
        alert("❌ Network Error");
      } finally {
        btn.textContent = originalText;
      }
    }
  </script>

    const video = document.getElementById('video');
    const canvas = document.getElementById('canvas');
    const btnSnap = document.getElementById('btnSnap');
    const btnSubmit = document.getElementById('btnSubmit');
    const instruction = document.getElementById('instruction');

    let step = 1; // 1=Front, 2=Left, 3=Right

    // Start Camera
    navigator.mediaDevices.getUserMedia({ video: { facingMode: "user" } })
        .then(stream => { video.srcObject = stream; })
        .catch(err => alert("Camera Error: " + err));

    async function captureStep() {
        // Disable button to prevent double clicks
        btnSnap.disabled = true;
        btnSnap.textContent = "⏳ Checking Quality...";
        
        const context = canvas.getContext('2d');
        canvas.width = video.videoWidth;
        canvas.height = video.videoHeight;
        context.drawImage(video, 0, 0);
        
        const dataURL = canvas.toDataURL('image/jpeg');

        // --- ⚡ SEND TO SERVER FOR CHECK ---
        try {
            const response = await fetch('/check_photo_quality', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ image: dataURL })
            });
            
            const result = await response.json();

            if (!result.valid) {
                // ❌ IF BAD PHOTO
                alert(result.error);
                btnSnap.disabled = false;
                btnSnap.textContent = "📸 Try Again";
                return; // Stop here! Don't move to next step.
            }

        } catch (err) {
            console.error(err);
            alert("Connection error. Please try again.");
            btnSnap.disabled = false;
            btnSnap.textContent = "📸 Capture";
            return;
        }

        // ✅ IF GOOD PHOTO -> PROCEED
        btnSnap.disabled = false;

        if(step === 1) {
            document.getElementById('input_front').value = dataURL;
            document.getElementById('thumb1').src = dataURL;
            document.getElementById('thumb1').classList.add('done');
            
            step = 2;
            instruction.textContent = "2. Turn Head Slightly LEFT";
            btnSnap.textContent = "📸 Capture Left Side";
            document.getElementById('dot2').classList.add('active');
            
        } else if(step === 2) {
            document.getElementById('input_left').value = dataURL;
            document.getElementById('thumb2').src = dataURL;
            document.getElementById('thumb2').classList.add('done');
            
            step = 3;
            instruction.textContent = "3. Turn Head Slightly RIGHT";
            btnSnap.textContent = "📸 Capture Right Side";
            document.getElementById('dot3').classList.add('active');

        } else if(step === 3) {
            document.getElementById('input_right').value = dataURL;
            document.getElementById('thumb3').src = dataURL;
            document.getElementById('thumb3').classList.add('done');
            
            instruction.textContent = "✅ All Angles Captured & Verified!";
            instruction.style.color = "#00ff88";
            btnSnap.style.display = "none"; 
            
            btnSubmit.disabled = false;
            btnSubmit.textContent = "📝 Register Account";
            btnSubmit.style.boxShadow = "0 0 20px #00ff88";
        }
    }

  <script>
    // 1. Initialize Week Dropdown (Current + 5 Past Weeks)
    function initWeeks() {
      const select = document.getElementById('weekSelect');
      select.innerHTML = '';
      
      let curr = new Date();
      // Adjust to Monday
      const day = curr.getDay();
      const diff = curr.getDate() - day + (day == 0 ? -6 : 1);
      curr.setDate(diff);

      for (let i = 0; i < 6; i++) {
        let start = new Date(curr);
        start.setDate(start.getDate() - (i * 7));
        
        let end = new Date(start);
        end.setDate(end.getDate() + 6); // End on Sunday
        
        // Format dates nicely (e.g., "Oct 23")
        let startStr = start.toLocaleDateString('en-US', { month: 'short', day: 'numeric' });
        let endStr = end.toLocaleDateString('en-US', { month: 'short', day: 'numeric' });
        
        let label = i === 0 ? "Current Week" : `Week ${i} Ago`;
        
        let option = document.createElement('option');
        option.value = start.toISOString().split('T')[0]; // Send YYYY-MM-DD to backend
        option.text = `${label} (${startStr} - ${endStr})`;
        select.appendChild(option);
      }
    }

    // 2. Main Fetch Function
    async function fetchData() {
      const sem = document.getElementById('semesterSelect').value;
      const sub = document.getElementById('subjectSelect').value;
      const startVal = document.getElementById('weekSelect').value;
      
      // Update Column Headers with Specific Dates
      updateHeaders(startVal);

      const tbody = document.getElementById('attendanceGrid');
      tbody.innerHTML = '<tr><td colspan="8" style="padding:40px; color:#888;">🔄 Loading attendance records...</td></tr>';

      try {
        const url = `/get_weekly_attendance?semester=${encodeURIComponent(sem)}&subject=${encodeURIComponent(sub)}&start_date=${startVal}`;
        const res = await fetch(url);
        const data = await res.json();

        tbody.innerHTML = '';
        
        if(data.length === 0) {
          tbody.innerHTML = '<tr><td colspan="8" style="padding:40px; color:#666;">No students found in this semester.</td></tr>';
          return;
        }

        // Base date for calculating circle logic
        let baseDate = new Date(startVal);

        data.forEach(student => {
          let daysHtml = '';
          const days = ['M','T','W','T','F','S'];
          
          // Generate 6 circles
          for(let i=0; i<6; i++) {
            let loopDate = new Date(baseDate);
            loopDate.setDate(loopDate.getDate() + i);
            let dateStr = loopDate.toISOString().split('T')[0];
            
            let record = student.week_data[dateStr];
            let status = record ? record.status : 'None';
            
            // Tooltip Text
            let tooltip = record 
              ? `${record.status} - ${record.time}` 
              : 'No Class';

            daysHtml += `<td>
              <div class="day-circle status-${status}" data-tooltip="${tooltip}">
                ${days[i]}
              </div>
            </td>`;
          }

          // Percent Logic
          let pct = student.overall_percent;
          let pctClass = pct >= 75 ? 'pct-high' : (pct >= 50 ? 'pct-mid' : 'pct-low');

          const tr = document.createElement('tr');
          tr.innerHTML = `
            <td style="text-align:left; padding-left:20px;">
              <div style="font-weight:bold; color:#fff; font-size:1.1em;">${student.name}</div>
              <div style="font-size:0.85em; color:#888;">${student.roll}</div>
            </td>
            ${daysHtml}
            <td><span class="pct-badge ${pctClass}">${pct}%</span></td>
          `;
          tbody.appendChild(tr);
        });

      } catch(e) {
        console.error(e);
        tbody.innerHTML = '<tr><td colspan="8" style="color:#ff4444;">❌ Error loading data from server</td></tr>';
      }
    }

    // 3. Helper to Update Table Headers (Mon 23 Oct)
    function updateHeaders(startDateStr) {
      let current = new Date(startDateStr);
      const days = ['Mon', 'Tue', 'Wed', 'Thu', 'Fri', 'Sat'];
      
      for(let i=0; i<6; i++) {
        // Format: "Oct 23"
        let datePretty = current.toLocaleDateString('en-US', { month: 'short', day: 'numeric' });
        
        // Update the TH text: "Mon <br> <small>Oct 23</small>"
        document.getElementById(`th_${i}`).innerHTML = `${days[i]} <small>${datePretty}</small>`;
        
        // Next day
        current.setDate(current.getDate() + 1);
      }
    }

    // Init
    initWeeks();
    fetchData();
